apiVersion: batch/v1
kind: Job
metadata:
  name: storage-reader-job
spec:
  template:
    spec:
      containers:
      - name: busybox
        image: busybox:1.36
        command: ["sh", "-c", "cat /data/hello.txt"]
        volumeMounts:
        - name: data-vol
          mountPath: /data
      volumes:
      - name: data-vol
        persistentVolumeClaim:
          claimName: localpath-hello-pvc
      restartPolicy: Never
  backoffLimit: 2
EOF



k8s-training-student-resources/Day_2/Lesson_5_csi_storage$
kubectl wait --for=condition=complete job/storage-reader-job --timeout=60s
